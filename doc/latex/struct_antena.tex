\doxysubsection{Detailed Description}
Estrutura que representa uma antena. 

Cada antena possui uma frequência, coordenadas (x, y) e um apontador para a próxima antena na lista.


\begin{DoxyParams}{Parameters}
{\em frequencia} & Frequência da antena (caractere). \\
\hline
{\em x} & Coordenada x da antena (inteiro não negativo). \\
\hline
{\em y} & Coordenada y da antena (inteiro não negativo). \\
\hline
{\em prox} & Apontador para a próxima antena na lista. \\
\hline
\end{DoxyParams}
